    <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
    <script>
        $(document).ready(function() {
            $('#filterInput').on('keyup', function() {
                var filterText = $(this).val().toUpperCase(); 

                $("#miRNATable tbody tr").each(function() {
                    var sequence = $(this).find("td:nth-child(3)").text(); 
                    if(sequence.toUpperCase().indexOf(filterText) > -1) {
                        $(this).removeClass("hiddenRow"); 
                    } else {
                        $(this).addClass("hiddenRow"); 
                    }
                });
            });

            
            $('#resetButton').click(function() {
                $('#filterInput').val(''); 
                $("#miRNATable tbody tr").show().removeClass("hiddenRow"); 
            });
        });

    </script>
